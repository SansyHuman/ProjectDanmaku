526  public static CartesianTimeFunction GetNaturalCubicSplineCurve(params UnityEngine.Vector2[] points)
527  {
528  int count = points.Length;
529  if (count == 0)
530  return new CartesianTimeFunction((t) => new CartesianCoord());
531 
532  float[] distances = new float[count - 1];
533  float totalLength = 0;
534  for (int i = 0; i < count - 1; i++)
535  {
536  distances[i] = (points[i + 1] - points[i]).magnitude;
537  totalLength += distances[i];
538  }
539 
540  float[] parameters = new float[count];
541  parameters[0] = 0;
542  for (int i = 0; i < count - 2; i++)
543  parameters[i + 1] = parameters[i] + (distances[i] / totalLength);
544  parameters[count - 1] = 1;
545 
546  float interval = 1f / (count - 1);
547 
548  (float t, float x)[] xPoints = new (float t, float x)[count];
549  (float t, float y)[] yPoints = new (float t, float y)[count];
550  for (int i = 0; i < count; i++)
551  {
552  xPoints[i] = (parameters[i], points[i].x);
553  yPoints[i] = (parameters[i], points[i].y);
554  }
555 
556  var xFunc = GetNaturalCubicSplineFunction(xPoints);
557  var yFunc = GetNaturalCubicSplineFunction(yPoints);
558 
559  return t => new CartesianCoord(xFunc(t), yFunc(t));
560  }

678  public static TimeFunction GetElasticOscillation(int oscillCount, int inclination)
679  {
680  TimeFunction oscillation = new TimeFunction((t) => Mathf.Sin(t * Mathf.PI)); // 0 <= t <= 2 * oscillTime - 0.5f
681  TimeFunction exponential = new TimeFunction((t) => GetExponential(4)(2 * t) * 2 - 1); // 0 <= t <= 0.5f
682  TimeFunction sum = new TimeFunction((t) =>
683  {
684  if (t < 2 * oscillCount - 0.5f)
685  return oscillation(t);
686  else
687  return exponential(t - (2 * oscillCount - 0.5f));
688  }).Composite((t) => t * 2 * oscillCount);
689  TimeFunction amplitude = GetExponential(inclination);
690  return new TimeFunction((t) => amplitude(t) * sum(t));
691  }

698  public static TimeFunction GetBounce(int bounceCount)
699  {
700  float[] amplitudes = new float[bounceCount];
701  float[] intervals = new float[bounceCount];
702  float[] intervalSum = new float[bounceCount + 1];
703  intervalSum[0] = 0;
704  for (int i = 0; i < bounceCount; i++)
705  {
706  amplitudes[i] = Mathf.Pow(2, i);
707  intervals[i] = Mathf.Pow(2, i);
708  intervalSum[i + 1] = intervalSum[i] + intervals[i];
709  if (i == bounceCount - 1)
710  intervalSum[i + 1] -= intervals[i] * 0.5f;
711  }
712 
713  TimeFunction sum = new TimeFunction((t) =>
714  {
715  if (t < 0)
716  return 0;
717  for (int i = 0; i < bounceCount; i++)
718  {
719  if (intervalSum[i] <= t && t < intervalSum[i + 1])
720  {
721  TimeFunction sin = new TimeFunction((q) => Mathf.Sin(q * Mathf.PI / intervals[i]) * amplitudes[i]);
722  return sin(t - intervalSum[i]);
723  }
724  }
725  return Mathf.Pow(2, bounceCount - 1);
726  });
727  return new TimeFunction((t) => sum(t) / Mathf.Pow(2, bounceCount - 1)).Composite((t) => t * intervalSum[bounceCount]);
728  }
