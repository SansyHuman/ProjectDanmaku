97  // Plays registered patterns in order.
98  private IEnumerator ManagePatterns()
99  {
100  for (int i = 0; i < shotPatterns.Count; i++)
101  {
102  currentPhase = i;
103  health = shotPatterns[i].health;
104  shotPatterns[i].shotPattern.gameObject.SetActive(true);
105  UDEBaseShotPattern pattern = shotPatterns[i].shotPattern;
106  pattern.Initialize(this);
107  pattern.StartPattern();
108  yield return new WaitUntil(() =>
109  (health < 0 || (pattern.HasTimeLimit && pattern.Time > pattern.TimeLimit)));
110  pattern.EndPattern();
111  OnPatternEnd(pattern);
112  }
113  alive = false;
114  OnDeath();
115  }
