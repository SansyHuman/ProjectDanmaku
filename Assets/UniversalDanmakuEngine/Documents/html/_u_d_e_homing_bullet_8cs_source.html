98  float distance = (this.position - enemyPosition).sqrMagnitude;
99  if (distance < closestDistance)
100  {
101  closestDistance = distance;
102  closestEnemy = i;
103  closestEnemyPosition = enemyPosition;
104  }
105  }
106 
107  float scaledDeltaTime = deltaTime * UDETime.Instance.PlayerTimeScale;
108 
109  if (closestEnemy >= 0)
110  {
111  float enemyAngle = UDEMath.Deg(closestEnemyPosition - this.position);
112  float extraHomingPower = enemyAngle > 90 ? (enemyAngle - 45) / 120f : 1;
113  movements[0].angle = Mathf.LerpAngle(movements[0].angle, enemyAngle, homingPower * extraHomingPower * scaledDeltaTime);
114  }
115 
116  float dx = movements[0].speed * Mathf.Cos(movements[0].angle * Mathf.Deg2Rad) * scaledDeltaTime;
117  float dy = movements[0].speed * Mathf.Sin(movements[0].angle * Mathf.Deg2Rad) * scaledDeltaTime;
118  Vector2 displacement = new Vector2(dx, dy);
119  this.position += displacement;
120  bulletTr.SetPositionAndRotation(this.position, Quaternion.Euler(0, 0, movements[0].angle));
121  time += scaledDeltaTime;
122  movements[0].updateCount++;
123  }
