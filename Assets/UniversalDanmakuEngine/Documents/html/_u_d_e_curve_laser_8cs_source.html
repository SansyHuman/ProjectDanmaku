138  protected override void ExtendLaser(float deltaTime)
139  {
140  base.ExtendLaser(deltaTime);
141 
142  float scaledDeltaTime = deltaTime * UDETime.Instance.GetTimeScale(timeScale);
143  time += scaledDeltaTime;
144 
145  if (laserHead == null)
146  goto PointsCheck;
147 
148  switch (mode)
149  {
150  case HeadMoveMode.CARTESIAN_FUNCTION:
151  laserHeadTr.localPosition = cartesianHeadFunction(time).ToVector2();
152  break;
153  case HeadMoveMode.POLAR_FUNCTION:
154  laserHeadTr.localPosition = ((UDEMath.CartesianCoord)polarHeadFunction(time)).ToVector2();
155  break;
156  }
157 
158  if ((Vector2)laserHeadTr.localPosition == points[points.Count - 1])
159  goto PointsCheck;
160 
161  points.Add(laserHeadTr.localPosition);
162  pointAddTimes.Add(time);
163 
164  PointsCheck:
165  for (int i = 0; i < points.Count; i++)
166  {
167  if (time - pointAddTimes[i] > pointDuration)
168  {
169  points.RemoveAt(i);
170  pointAddTimes.RemoveAt(i);
171  i--;
172  }
173  else
174  break;
175  }
176 
177  laser.positionCount = points.Count;
178  for (int i = 0; i < points.Count; i++)
179  laser.SetPosition(i, points[i]);
180  laserCollider.points = points.ToArray();
181 
182  if (time > headDuration)
183  {
184  Destroy(laserHead);
185  laserHead = null;
186  }
187 
188  if (points.Count == 0)
189  DestroyLaser();
190  }
191  }
