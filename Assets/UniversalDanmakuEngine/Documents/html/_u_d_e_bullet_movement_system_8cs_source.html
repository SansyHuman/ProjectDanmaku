101  float3 radius = translation.Value - origin;

103  polarCoord.Angle = polarCoord.Radius > 0.01f ? UDEMath.Deg(radius.x, radius.y) : rotation.Value.z;

108  // Calculates the displacement of the bullet in cartesian mode.
109  private static float2 CartesianMove(float deltaTime, ref UDEBulletMovementECS movement, ref RotationEulerXYZ rot)
110  {
111  float2 dr = movement.velocity * deltaTime;
112  if (movement.faceToMovingDirection)
113  rot.Value.z = math.radians(UDEMath.Deg(movement.velocity));
114  else
115  SetRotation(deltaTime, ref movement, ref rot);
116 
117  movement.velocity += movement.accel * deltaTime;
118  if (movement.limitSpeed)
119  {
120  float speed = math.length(movement.velocity);
121  if (speed > movement.maxMagnitude)
122  movement.velocity = movement.velocity * movement.maxMagnitude / speed;
123  if (speed < movement.minMagnitude)
124  movement.velocity = movement.velocity * movement.minMagnitude / speed;
125 
126  if (movement.velocity.x > movement.maxVelocity.x)
127  movement.velocity.x = movement.maxVelocity.x;
128  if (movement.velocity.y > movement.maxVelocity.y)
129  movement.velocity.y = movement.maxVelocity.y;
130  if (movement.velocity.x < movement.minVelocity.x)
131  movement.velocity.x = movement.minVelocity.x;
132  if (movement.velocity.y < movement.minVelocity.y)
133  movement.velocity.y = movement.minVelocity.y;
134  }
135  return dr;
136  }

138  // Calculates the displacement of the bullet in cartesian-polar mode.
139  private static float2 CartesianPolarMove(float deltaTime, ref UDEBulletMovementECS movement, ref RotationEulerXYZ rot)
140  {
141  float dx = movement.speed * math.cos(math.radians(movement.angle)) * deltaTime;
142  float dy = movement.speed * Mathf.Sin(math.radians(movement.angle)) * deltaTime;
143  if (movement.faceToMovingDirection)
144  rot.Value.z = math.radians(movement.angle);
145  else
146  SetRotation(deltaTime, ref movement, ref rot);
147 
148  movement.speed += movement.tangentialAccel * deltaTime;
149  movement.angle += movement.normalAccel * deltaTime;
150  while (movement.angle > 360)
151  movement.angle -= 360;
152  while (movement.angle < 0)
153  movement.angle += 360;
154  if (movement.limitSpeed)
155  {
156  if (movement.speed > movement.maxSpeed)
157  movement.speed = movement.maxSpeed;
158  if (movement.speed < movement.minSpeed)
159  movement.speed = movement.minSpeed;
160  }
161 
162  return new float2(dx, dy);
163  }

165  // Calculates the displacement of the bullet in polar coordinate system.
166  private static float2 PolarMove(float deltaTime, ref UDEBulletMovementECS movement, ref RotationEulerXYZ rot, ref UDEPolarCoordinate polar)
167  {
168  float xPre, yPre;
169  UDEMath.Polar2Cartesian(polar.Radius, polar.Angle, out xPre, out yPre);
170  float2 rPre = new float2(xPre, yPre);
171 
172  polar.Radius += movement.radialSpeed * deltaTime;
173  movement.radialSpeed += movement.radialAccel * deltaTime;
174  polar.Angle += movement.angularSpeed * deltaTime;
175  movement.angularSpeed += movement.angularAccel * deltaTime;
176  while (polar.Angle > 360)
177  polar.Angle -= 360;
178  while (polar.Angle < 0)
179  polar.Angle += 360;
180 
181  if (movement.limitSpeed)
182  {
183  if (movement.radialSpeed > movement.maxRadialSpeed)
184  movement.radialSpeed = movement.maxRadialSpeed;
185  if (movement.radialSpeed < movement.minRadialSpeed)
186  movement.radialSpeed = movement.minRadialSpeed;
187  if (movement.angularSpeed > movement.maxAngularSpeed)
188  movement.angularSpeed = movement.maxAngularSpeed;
189  if (movement.angularSpeed < movement.minAngularSpeed)
190  movement.angularSpeed = movement.minAngularSpeed;
191  }
192  float x, y;
193  UDEMath.Polar2Cartesian(polar.Radius, polar.Angle, out x, out y);
194  movement.angle = polar.Angle;
195 
196  float2 displacement = new float2(x, y) - rPre;
197  if (movement.faceToMovingDirection)
198  rot.Value.z = math.lengthsq(displacement) > 0.0001f ? math.radians(UDEMath.Deg(displacement)) : rot.Value.z;
199  else
200  SetRotation(deltaTime, ref movement, ref rot);
201 
202  return displacement;
203  }

205  // Sets rotation if faceToMovingDirection in UDEBulletMovement is false.
206  private static void SetRotation(float deltaTime, ref UDEBulletMovementECS movement, ref RotationEulerXYZ rotation)
207  {
208  float newRot = math.degrees(rotation.Value.z);
209  newRot += movement.rotationAngularSpeed * deltaTime;
210  while (newRot > 360)
211  newRot -= 360;
212  while (newRot < 0)
213  newRot += 360;
214 
215  movement.rotationAngularSpeed += movement.rotationAngularAcceleration * deltaTime;
216  if (movement.limitRotationSpeed)
217  {
218  if (movement.rotationAngularSpeed > movement.maxRotationSpeed)
219  movement.rotationAngularSpeed = movement.maxRotationSpeed;
220  if (movement.rotationAngularSpeed < movement.minRotationSpeed)
221  movement.rotationAngularSpeed = movement.minRotationSpeed;
222  }
223  rotation.Value.z = math.radians(newRot);
224  }

240  public void Execute(Entity entity, int index, ref UDEBulletMovements movement, ref UDEPolarCoordinate polarCoord)
241  {
242  UDEBulletECS bullet = manager.GetComponentObject<UDEBulletECS>(entity);
243  int phase = movement.Phase;
244  while (!(phase >= movement.TotalPhase - 1) && movement.Time >= bullet.movements[phase + 1].startTime)
245  {
246  phase++;
247  }
248  if (phase != movement.Phase)
249  {
250  UDEBulletMovementECS prevMovement = movement.Movement;
251  movement.Movement = bullet.movements[phase];
252  if (movement.Movement.setSpeedToPrevMovement)
253  SyncMovementSpeed(ref prevMovement, ref movement.Movement, ref polarCoord);
254  movement.Phase = phase;
255  }
256 
257  if (movement.Movement.hasEndTime && movement.Time >= movement.Movement.endTime)
258  {
259  if (movement.Phase == movement.TotalPhase - 1 && movement.Loop)
260  {
261  UDEBulletMovementECS prevMovement = movement.Movement;
262  movement.Movement = bullet.movements[0];
263  if (movement.Movement.setSpeedToPrevMovement)
264  SyncMovementSpeed(ref prevMovement, ref movement.Movement, ref polarCoord);
265  movement.Phase = 0;
266  movement.Time = 0;
267  }

279  internal static void SyncMovementSpeed(ref UDEBulletMovementECS prev, ref UDEBulletMovementECS next, ref UDEPolarCoordinate polarCoord)
280  {
281  float angle = polarCoord.Angle;
282  Vector2 prevVelocity = new Vector2(0, 0);
283  switch (prev.mode)
284  {
285  case MoveMode.CARTESIAN:
286  prevVelocity = prev.velocity;
287  break;
288  case MoveMode.CARTESIAN_POLAR:
289  var velocityTuple = UDEMath.Polar2Cartesian(prev.speed, prev.angle);
290  prevVelocity = new Vector2(velocityTuple.x, velocityTuple.y);
291  break;
292  case MoveMode.POLAR:
293  var radialVel = UDEMath.Polar2Cartesian(prev.radialSpeed, angle);
294  var angularVel = UDEMath.Polar2Cartesian(prev.angularSpeed * Mathf.Deg2Rad * polarCoord.Radius, angle + 90f);
295  prevVelocity = new Vector2(radialVel.x + angularVel.x, radialVel.y + angularVel.y);
296  break;
297  }
298 
299  switch (next.mode)
300  {
301  case MoveMode.CARTESIAN:
302  next.velocity = prevVelocity;
303  break;
304  case MoveMode.CARTESIAN_POLAR:
305  (float speed, float angle) polarVel = UDEMath.Cartesian2Polar(prevVelocity);
306  next.speed = polarVel.speed;
307  next.angle = polarVel.angle;
308  break;
309  case MoveMode.POLAR:
310  (float speed, float angle) polarVel2 = UDEMath.Cartesian2Polar(prevVelocity);
311  next.radialSpeed = polarVel2.speed * Mathf.Cos((polarVel2.angle - angle) * Mathf.Deg2Rad);
312  next.angularSpeed = polarVel2.speed * Mathf.Sin((polarVel2.angle - angle) * Mathf.Deg2Rad) * Mathf.Rad2Deg / polarCoord.Radius;
313  break;
314  }
315  }
