105  public TransitionResult AddInfo(ITransitiontInfo info)
106  {
107  TransitionInfo information = new TransitionInfo();
108  information.info = info;
109  TransitionResult result = new TransitionResult();
110  information.result = result;
111 
112  if (information.info.IsPhysics)
113  physicsInfos.Add(information);
114  else
115  nonphysicsInfos.Add(information);
116 
117  if (!enabled)
118  enabled = true;
119 
120  return result;
121  }
122 
123  // Not physics
124  private void Update()
125  {
126  if (nonphysicsInfos.Count == 0)
127  {
128  if (physicsInfos.Count == 0)
129  enabled = false;
130 
131  return;
132  }
133 
134  ITransitiontInfo current;
135  for (int i = 0; i < nonphysicsInfos.Count; i++)
136  {
137  current = nonphysicsInfos[i].info;
138  current.Progress(Time.deltaTime, tr);
139  if (current.Progression >= 1)
140  {
141  nonphysicsInfos[i].result.endTransition = true;
142  nonphysicsInfos.RemoveAt(i);
143  i--;
144  }
145  }
146  }
147 
148  // Physics
149  private void FixedUpdate()
150  {
151  if (physicsInfos.Count == 0)
152  {
153  if (nonphysicsInfos.Count == 0)
154  enabled = false;
155 
156  return;
157  }
158 
159  ITransitiontInfo current;
160  for (int i = 0; i < physicsInfos.Count; i++)
161  {

164  if (current.Progression >= 1)
165  {
166  physicsInfos[i].result.endTransition = true;

169  }
170  }
171  }

201  public void Progress(float deltaTime, Transform transform)
202  {
203  float scaledDeltaTime = deltaTime;
204  switch (timeScale)
205  {
206  case UDETime.TimeScale.ENEMY:
207  scaledDeltaTime *= UDETime.Instance.EnemyTimeScale;
208  break;
209  case UDETime.TimeScale.PLAYER:
210  scaledDeltaTime *= UDETime.Instance.PlayerTimeScale;
211  break;
212  case UDETime.TimeScale.UNSCALED:
213  default:
214  break;
215  }
216 
217  time += scaledDeltaTime;
218  if (time > duration)
219  time = duration;
220 
221  progression = time / duration;
222  float t = easeFunction(progression);
223  Vector2 location = from * (1 - t) + to * t;
224 
225  transform.position = location;
226  }

451  public static UDEMath.TimeFunction ConvertIn2Out(UDEMath.TimeFunction easeIn)
458  {
459  return new UDEMath.TimeFunction((t) => 1 - easeIn(1 - t));
460  }
461 
468  public static UDEMath.TimeFunction CombineInOutEase(UDEMath.TimeFunction easeIn, UDEMath.TimeFunction easeOut)
469  {
470  UDEMath.TimeFunction doubleInterval = new UDEMath.TimeFunction((t) => 2 * t);
471  UDEMath.TimeFunction comb = new UDEMath.TimeFunction((t) =>
472  {
473  return t < 1 ? 0.5f * easeIn(t) : 0.5f + 0.5f * easeOut(t - 1);
474  });
475  return comb.Composite(doubleInterval);
476  }
