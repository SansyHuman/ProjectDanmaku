105  public TransitionResult AddInfo(ITransitiontInfo info)
106  {
107  TransitionInfo information = new TransitionInfo();
108  information.info = info;
109  TransitionResult result = new TransitionResult();
110  information.result = result;
111 
112  if (information.info.IsPhysics)
113  physicsInfos.Add(information);
114  else
115  nonphysicsInfos.Add(information);
116 
117  if (!enabled)
118  enabled = true;
119 
120  return result;
121  }

128  nonphysicsInfos.Clear();
129  physicsInfos.Clear();

131  }

133  // Not physics
134  private void Update()
135  {
136  if (nonphysicsInfos.Count == 0)
137  {
138  if (physicsInfos.Count == 0)
139  enabled = false;
140 
141  return;
142  }
143 
144  ITransitiontInfo current;
145  for (int i = 0; i < nonphysicsInfos.Count; i++)
146  {
147  current = nonphysicsInfos[i].info;
148  current.Progress(Time.deltaTime, tr);
149  if (current.Progression >= 1)
150  {
151  nonphysicsInfos[i].result.endTransition = true;
152  nonphysicsInfos.RemoveAt(i);
153  i--;
154  }
155  }
156  }
157

211  public void Progress(float deltaTime, Transform transform)
212  {
213  float scaledDeltaTime = deltaTime;
214  switch (timeScale)
215  {
216  case UDETime.TimeScale.ENEMY:
217  scaledDeltaTime *= UDETime.Instance.EnemyTimeScale;
218  break;
219  case UDETime.TimeScale.PLAYER:
220  scaledDeltaTime *= UDETime.Instance.PlayerTimeScale;
221  break;
222  case UDETime.TimeScale.UNSCALED:
223  default:
224  break;
225  }
226 
227  time += scaledDeltaTime;
228  if (time > duration)
229  time = duration;
230 
231  progression = time / duration;
232  float t = easeFunction(progression);
233  Vector2 location = from * (1 - t) + to * t;
234 
235  transform.position = location;
236  }

461  public static UDEMath.TimeFunction ConvertIn2Out(UDEMath.TimeFunction easeIn)
468  {
469  return new UDEMath.TimeFunction((t) => 1 - easeIn(1 - t));
470  }
471 
478  public static UDEMath.TimeFunction CombineInOutEase(UDEMath.TimeFunction easeIn, UDEMath.TimeFunction easeOut)
479  {
480  UDEMath.TimeFunction doubleInterval = new UDEMath.TimeFunction((t) => 2 * t);
481  UDEMath.TimeFunction comb = new UDEMath.TimeFunction((t) =>
482  {
483  return t < 1 ? 0.5f * easeIn(t) : 0.5f + 0.5f * easeOut(t - 1);
484  });
485  return comb.Composite(doubleInterval);
486  }
