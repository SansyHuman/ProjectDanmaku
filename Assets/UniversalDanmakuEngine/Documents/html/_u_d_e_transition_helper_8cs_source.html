105  public TransitionResult AddInfo(ITransitiontInfo info)
106  {
107  TransitionInfo information = new TransitionInfo();
108  information.info = info;
109  TransitionResult result = new TransitionResult();
110  information.result = result;
111 
112  if (information.info.IsPhysics)
113  physicsInfos.Add(information);
114  else
115  nonphysicsInfos.Add(information);
116 
117  if (!enabled)
118  enabled = true;
119 
120  return result;
121  }
122 
123  // Not physics
124  private void Update()
125  {
126  if (nonphysicsInfos.Count == 0)
127  {
128  if (physicsInfos.Count == 0)
129  enabled = false;
130 
131  return;
132  }
133 
134  ITransitiontInfo current;
135  for (int i = 0; i < nonphysicsInfos.Count; i++)
136  {
137  current = nonphysicsInfos[i].info;
138  current.Progress(Time.deltaTime, tr);
139  if (current.Progression >= 1)
140  {
141  nonphysicsInfos[i].result.endTransition = true;
142  nonphysicsInfos.RemoveAt(i);
143  i--;
144  }
145  }
146  }
147 
148  // Physics

200  public void Progress(float deltaTime, Transform transform)
201  {
202  float scaledDeltaTime = deltaTime;
203  switch (timeScale)
204  {
205  case UDETime.TimeScale.ENEMY:
206  scaledDeltaTime *= UDETime.Instance.EnemyTimeScale;
207  break;
208  case UDETime.TimeScale.PLAYER:
209  scaledDeltaTime *= UDETime.Instance.PlayerTimeScale;
210  break;
211  case UDETime.TimeScale.UNSCALED:
212  default:
213  break;
214  }
215 
216  time += scaledDeltaTime;
217  if (time > duration)
218  time = duration;
219 
220  progression = time / duration;
221  float t = easeFunction(progression);
222  Vector2 location = from * (1 - t) + to * t;
223 
224  transform.position = location;
225  }

298  public static UDEMath.TimeFunction CombineInOutEase(UDEMath.TimeFunction easeIn, UDEMath.TimeFunction easeOut)
299  {
300  UDEMath.TimeFunction doubleInterval = new UDEMath.TimeFunction((t) => 2 * t);
301  UDEMath.TimeFunction comb = new UDEMath.TimeFunction((t) =>
302  {
303  return t < 1 ? 0.5f * easeIn(t) : 0.5f + 0.5f * easeOut(t - 1);
304  });
305  return comb.Composite(doubleInterval);
306  }
