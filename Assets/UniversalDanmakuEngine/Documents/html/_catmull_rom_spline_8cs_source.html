32  internal class CatmullRomSpline
33  {

36  private float tRange;
37  private int numPt;
38 
39  private readonly float alpha = 0.5f;
40 
41  internal CatmullRomSpline(params Vector2[] points)
42  {
43  if (points.Length < 4)
44  throw new ArgumentException("The number of points is less than 4.", nameof(points));
45  this.points = points;
46  numPt = this.points.Length;
47 
48  t = new float[numPt];
49  t[0] = 0;
50  for (int i = 1; i < numPt; i++)
51  t[i] = t[i - 1] + Pow(Sqrt((points[i] - points[i - 1]).sqrMagnitude), alpha);
52 
53  tRange = t[numPt - 2] - t[1];
54  }
55 
56  internal CartesianCoord Eval(float t) // t is between 0 and 1.
57  {
58  if (t < 0 || t > 1)
59  throw new ArgumentException("t is not in range", nameof(t));
60 
61  float realT = this.t[1] + t * tRange;
62  int zero = 0;
63  for (int i = 2; i < numPt - 2; i++)
64  {
65  if (realT <= this.t[i])
66  break;
67 
68  zero++;
69  }
70 
71  int one = zero + 1;
72  int two = zero + 2;
73  int three = zero + 3;
74 
75  float int10 = this.t[one] - this.t[zero];
76  float int21 = this.t[two] - this.t[one];
77  float int32 = this.t[three] - this.t[two];
78 
79  Vector2 A1 = (this.t[one] - realT) * points[zero] + (realT - this.t[zero]) * points[one];
80  A1 /= int10;
81  Vector2 A2 = (this.t[two] - realT) * points[one] + (realT - this.t[one]) * points[two];
82  A2 /= int21;
83  Vector2 A3 = (this.t[three] - realT) * points[two] + (realT - this.t[two]) * points[three];
84  A3 /= int32;
85 
86  float int20 = this.t[two] - this.t[zero];
87  float int31 = this.t[three] - this.t[one];
88 
89  Vector2 B1 = (this.t[two] - realT) * A1 + (realT - this.t[zero]) * A2;
90  B1 /= int20;
91  Vector2 B2 = (this.t[three] - realT) * A2 + (realT - this.t[one]) * A3;
92  B2 /= int31;
93 
94  Vector2 C = (this.t[two] - realT) * B1 + (realT - this.t[one]) * B2;
95  C /= int21;
96 
97  return new CartesianCoord(C.x, C.y);
98  }
