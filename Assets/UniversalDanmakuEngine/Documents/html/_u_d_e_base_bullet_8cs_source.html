249  protected virtual void MoveBullet(float deltaTime)
257  {
258  float scaledDeltaTime;
259  switch (usingTimeScale)
260  {
261  case UDETime.TimeScale.ENEMY:
262  scaledDeltaTime = deltaTime * UDETime.Instance.EnemyTimeScale;
263  break;
264  case UDETime.TimeScale.PLAYER:
265  scaledDeltaTime = deltaTime * UDETime.Instance.PlayerTimeScale;
266  break;
267  case UDETime.TimeScale.UNSCALED:
268  default:
269  scaledDeltaTime = deltaTime;
270  break;
271  }
272  time += scaledDeltaTime;
273 
274  if (halo != null)
275  {
276  if (isSummoning)
277  {
278  if (time > summonTime)
279  {
280  renderer.enabled = true;
281  collider.enabled = true;
282  halo.enabled = false;
283  isSummoning = false;
284  time = 0;
285  }
286  }
287  }
288 
289  int prevPhase = phase;
290  while (!(phase >= movements.Length - 1) && time >= movements[phase + 1].startTime)
291  {
292  phase++;
293  }
294  if (phase != prevPhase)
295  {
296  movements[phase] = movementsOriginal[phase];
297  if (movements[phase].setSpeedToPrevMovement)
298  SyncMovementSpeed(ref movements[prevPhase], ref movements[phase]);
299  }
300  if (movements[phase].hasEndTime && time >= movements[phase].endTime)
301  {
302  if (phase == movements.Length - 1 && loop)
303  {
304  movements[0] = movementsOriginal[0];
305  if (movements[0].setSpeedToPrevMovement)
306  SyncMovementSpeed(ref movements[phase], ref movements[0]);
307  phase = 0;
308  }
309  else
310  return;
311  }
312  if (phase == 0 && time < movements[phase].startTime)
313  return;
314 
315  ref UDEBulletMovement currentMovement = ref movements[phase];
316  Vector2 displacement;
317  switch (currentMovement.mode)
318  {
319  case MoveMode.CARTESIAN:
320  displacement = CartesianMove(scaledDeltaTime, ref currentMovement);
321  break;
322  case MoveMode.CARTESIAN_POLAR:
323  displacement = CartesianPolarMove(scaledDeltaTime, ref currentMovement);
324  break;
325  case MoveMode.POLAR:
326  displacement = PolarMove(scaledDeltaTime, ref currentMovement);
327  break;
328  default:
329  displacement = new Vector2(0, 0);
330  break;
331  }
332 
333  position += displacement;
334  if (currentMovement.mode != MoveMode.POLAR)
335  {
336  if (setOriginToCharacter)
337  origin = originChrTr.position;
338  r = (position - origin).magnitude;
339  angle = r > 0.01f ? UDEMath.Deg(position - origin) : rotation;
340  }
341  bulletTr.SetPositionAndRotation(position, Quaternion.Euler(0, 0, rotation));
342  currentMovement.updateCount++;
343  }

350  protected void SyncMovementSpeed(ref UDEBulletMovement prev, ref UDEBulletMovement next)
351  {
352  Vector2 prevVelocity = new Vector2(0, 0);
353  switch (prev.mode)
354  {
355  case MoveMode.CARTESIAN:
356  prevVelocity = prev.velocity;
357  break;
358  case MoveMode.CARTESIAN_POLAR:
359  var velocityTuple = UDEMath.Polar2Cartesian(prev.speed, prev.angle);
360  prevVelocity = new Vector2(velocityTuple.x, velocityTuple.y);
361  break;
362  case MoveMode.POLAR:
363  var radialVel = UDEMath.Polar2Cartesian(prev.radialSpeed, this.angle);
364  var angularVel = UDEMath.Polar2Cartesian(prev.angularSpeed, this.angle + 90f);
365  prevVelocity = new Vector2(radialVel.x + angularVel.x, radialVel.y + angularVel.y);
366  break;
367  }
368 
369  switch (next.mode)
370  {
371  case MoveMode.CARTESIAN:
372  next.velocity = prevVelocity;
373  break;
374  case MoveMode.CARTESIAN_POLAR:
375  (float speed, float angle) polarVel = UDEMath.Cartesian2Polar(prevVelocity);
376  next.speed = polarVel.speed;
377  next.angle = polarVel.angle;
378  break;
379  case MoveMode.POLAR:
380  (float speed, float angle) polarVel2 = UDEMath.Cartesian2Polar(prevVelocity);
381  next.radialSpeed = polarVel2.speed * Mathf.Cos(polarVel2.angle - this.angle);
382  next.angularSpeed = polarVel2.speed * Mathf.Sin(polarVel2.angle - this.angle);
383  break;
384  }
385  }

387  // Calculates the displacement of the bullet in cartesian mode.
388  private Vector2 CartesianMove(float deltaTime, ref UDEBulletMovement movement)
389  {
390  Vector2 dr = movement.velocity * deltaTime;
391  if (movement.faceToMovingDirection)
392  rotation = UDEMath.Deg(movement.velocity);
393  else
394  SetRotation(deltaTime, ref movement);
395 
396  float passedTime = time - movement.startTime;
397  movement.velocity += (Vector2)movement.accel(passedTime) * deltaTime;
398  if (movement.limitSpeed)
399  {
400  if (movement.velocity.magnitude > movement.maxMagnitude)
401  movement.velocity = movement.velocity * movement.maxMagnitude / movement.velocity.magnitude;
402  if (movement.velocity.magnitude < movement.minMagnitude)
403  movement.velocity = movement.velocity * movement.minMagnitude / movement.velocity.magnitude;
404  if (movement.velocity.x > movement.maxVelocity.x)
405  movement.velocity.x = movement.maxVelocity.x;
406  if (movement.velocity.y > movement.maxVelocity.y)
407  movement.velocity.y = movement.maxVelocity.y;
408  if (movement.velocity.x < movement.minVelocity.x)
409  movement.velocity.x = movement.minVelocity.x;
410  if (movement.velocity.y < movement.minVelocity.y)
411  movement.velocity.y = movement.minVelocity.y;
412  }
413  return dr;
414  }

416  // Calculates the displacement of the bullet in cartesian-polar mode.
417  private Vector2 CartesianPolarMove(float deltaTime, ref UDEBulletMovement movement)
418  {
419  float dx = movement.speed * Mathf.Cos(movement.angle * Mathf.Deg2Rad) * deltaTime;
420  float dy = movement.speed * Mathf.Sin(movement.angle * Mathf.Deg2Rad) * deltaTime;
421  if (movement.faceToMovingDirection)
422  rotation = movement.angle;
423  else
424  SetRotation(deltaTime, ref movement);
425 
426  float passedTime = time - movement.startTime;
427  movement.speed += movement.tangentialAccel(passedTime) * deltaTime;
428  movement.angle += movement.normalAccel(passedTime) * deltaTime;
429  while (movement.angle > 360)
430  movement.angle -= 360;
431  while (movement.angle < 0)
432  movement.angle += 360;
433  if (movement.limitSpeed)
434  {
435  if (movement.speed > movement.maxSpeed)
436  movement.speed = movement.maxSpeed;
437  if (movement.speed < movement.minSpeed)
438  movement.speed = movement.minSpeed;
439  }
440 
441  return new Vector2(dx, dy);
442  }

444  // Calculates the displacement of the bullet in polar coordinate system.
445  private Vector2 PolarMove(float deltaTime, ref UDEBulletMovement movement)
446  {
447  float xPre, yPre;
448  UDEMath.Polar2Cartesian(r, angle, out xPre, out yPre);
449  Vector2 rPre = new Vector2(xPre, yPre);
450 
451  float passedTime = time - movement.startTime;
452  r += movement.radialSpeed * deltaTime;
453  movement.radialSpeed += movement.radialAccel(passedTime) * deltaTime;
454  angle += movement.angularSpeed * deltaTime;
455  movement.angularSpeed += movement.angularAccel(passedTime) * deltaTime;
456  while (angle > 360)
457  angle -= 360;
458  while (angle < 0)
459  angle += 360;
460 
461  if (movement.limitSpeed)
462  {
463  if (movement.radialSpeed > movement.maxRadialSpeed)
464  movement.radialSpeed = movement.maxRadialSpeed;
465  if (movement.radialSpeed < movement.minRadialSpeed)
466  movement.radialSpeed = movement.minRadialSpeed;
467  if (movement.angularSpeed > movement.maxAngularSpeed)
468  movement.angularSpeed = movement.maxAngularSpeed;
469  if (movement.angularSpeed < movement.minAngularSpeed)
470  movement.angularSpeed = movement.minAngularSpeed;
471  }
472  float x, y;
473  UDEMath.Polar2Cartesian(r, angle, out x, out y);
474  movement.angle = angle;
475 
476  Vector2 originDisplacement = Vector2.zero;
477  if (setOriginToCharacter && originCharacter != null)
478  {
479  originDisplacement = originCharacter.transform.position - new Vector3(origin.x, origin.y);
480  origin = originChrTr.position;
481  }
482 
483  Vector2 displacement = new Vector2(x, y) - rPre;
484  if (movement.faceToMovingDirection)
485  rotation = displacement.sqrMagnitude > 0.0001f ? UDEMath.Deg(displacement) : rotation;
486  else
487  SetRotation(deltaTime, ref movement);
488 
489  return displacement + originDisplacement;
490  }

492  // Sets rotation if faceToMovingDirection in UDEBulletMovement is false.
493  private void SetRotation(float deltaTime, ref UDEBulletMovement movement)
494  {
495  rotation += movement.rotationAngularSpeed * deltaTime;
496  while (rotation > 360)
497  rotation -= 360;
498  while (rotation < 0)
499  rotation += 360;
500 
501  float passedTime = time - movement.startTime;
502  movement.rotationAngularSpeed += movement.rotationAngularAcceleration(passedTime) * deltaTime;
503  if (movement.limitRotationSpeed)
504  {
505  if (movement.rotationAngularSpeed > movement.maxRotationSpeed)
506  movement.rotationAngularSpeed = movement.maxRotationSpeed;
507  if (movement.rotationAngularSpeed < movement.minRotationSpeed)
508  movement.rotationAngularSpeed = movement.minRotationSpeed;
509  }
510  }

525  public void ForceMoveToPhase(int phase)
526  {
527  if (this.phase == phase || phase > movements.Length - 1 || phase < 0)
528  return;
529 
530  if (this.phase <= phase)
531  for (int i = this.phase + 1; i <= phase; i++)
532  movements[i] = movementsOriginal[i];
533  else
534  for (int i = this.phase - 1; i >= phase; i--)
535  movements[i] = movementsOriginal[i];
536 
537  if (movements[phase].setSpeedToPrevMovement)
538  SyncMovementSpeed(ref movements[this.phase], ref movements[phase]);
539 
540  movements[this.phase] = movementsOriginal[this.phase];
541 
542  time = movements[phase].startTime;
543  this.phase = phase;
544  }
