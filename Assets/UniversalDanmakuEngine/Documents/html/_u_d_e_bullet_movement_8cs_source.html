281  public override string ToString()
282  {
283  StringBuilder str = new StringBuilder("Move Mode: ");
284  str.Append(mode);
285  str.Append("\nStart Time: ").Append(startTime);
286  if (hasEndTime)
287  str.Append("\nEnd Time: ").Append(endTime);
288  switch (this.mode)
289  {
290  case MoveMode.CARTESIAN:
291  str.Append("\n\nVelocity: ").Append(velocity);
292  if (limitSpeed)
293  {
294  str.Append("\nMinimum x velocity: ").Append(minVelocity.x);
295  str.Append("\nMinimum y velocity: ").Append(minVelocity.y);
296  str.Append("\nMaximum x velocity: ").Append(maxVelocity.x);
297  str.Append("\nMaximum y velocity: ").Append(maxVelocity.y);
298  str.Append("\nMinimum speed: ").Append(minMagnitude);
299  str.Append("\nMaximum speed: ").Append(maxMagnitude);
300  }
301  break;
302  case MoveMode.CARTESIAN_POLAR:
303  str.Append("\n\nSpeed: ").Append(speed);
304  if (limitSpeed)
305  {
306  str.Append("\nMinimum Speed: ").Append(minSpeed);
307  str.Append("\nMaximum Speed: ").Append(maxSpeed);
308  }
309  str.Append("\nMoving Angle: ").Append(angle);
310  break;
311  case MoveMode.POLAR:
312  str.Append("\n\nRadial Speed: ").Append(radialSpeed);
313  str.Append("\nAngular Speed: ").Append(angularSpeed);
314  if (limitSpeed)
315  {
316  str.Append("\nMinimum Radial Speed: ").Append(minRadialSpeed);
317  str.Append("\nMaximum Radial Speed: ").Append(maxRadialSpeed);
318  str.Append("\nMinimum Angular Speed: ").Append(minAngularSpeed);
319  str.Append("\nMaximum Angular Speed: ").Append(maxAngularSpeed);
320  }
321  break;
322  }
323  if (faceToMovingDirection)
324  str.Append("\n\nThe bullet is facing to moving direction.");
325  else
326  {
327  str.Append("\n\nRotational Angular Speed: ").Append(rotationAngularSpeed);
328  if (limitRotationSpeed)
329  {
330  str.Append("\nMinimum Rotational Speed: ").Append(minRotationSpeed);
331  str.Append("\nMaximum Rotational Speed: ").Append(maxRotationSpeed);
332  }
333  }
334  return str.ToString();
335  }
336  }
