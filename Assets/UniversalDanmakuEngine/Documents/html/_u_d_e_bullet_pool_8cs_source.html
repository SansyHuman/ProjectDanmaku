83  public void AddNewPool(UDEAbstractBullet prefab, int initialNumber)
84  {
85  try
86  {
87  _ = poolList[prefab];
88  }
89  catch (KeyNotFoundException) // When the pool of the prefab does not exist.
90  {
91  Stack<UDEAbstractBullet> pool = new Stack<UDEAbstractBullet>(initialNumber * 2);
92  for (int j = 0; j < initialNumber; j++)
93  {
94  UDEAbstractBullet instance = Instantiate<UDEAbstractBullet>(prefab, new Vector3(0, 0, 0), Quaternion.Euler(0, 0, 0));
95 #pragma warning disable CS0618 // Type or member is obsolete

118  public void ClearPool(UDEAbstractBullet prefab, bool deleteActiveBullets = false)
119  {
120  Stack<UDEAbstractBullet> pool;
121  if (!poolList.TryGetValue(prefab, out pool))
122  {
123  Debug.LogWarning("The pool of the prefab does not exist. Pool deletion will be ignored.");
124  return;
125  }
126 
127  Stack<UDEAbstractBullet> temp = new Stack<UDEAbstractBullet>(pool.Count);
128 
129  for (int i = 0; i < pool.Count; i++)
130  {
131  UDEAbstractBullet bullet = pool.Pop();
132  if (!deleteActiveBullets && bullet.gameObject.activeSelf)
133  temp.Push(bullet);
134  else
135  Destroy(bullet.gameObject);
136  }
137 
138  if (temp.Count > 0)
139  {
140  for (int i = 0; i < temp.Count; i++)
141  pool.Push(temp.Pop());
142  }
143  else
144  {
145  poolList.Remove(prefab);
146  }
147  }

167  public UDEAbstractBullet GetBullet(UDEAbstractBullet target)
168  {
169  UDEAbstractBullet returnInstance = null;
170 
171  Stack<UDEAbstractBullet> targetPool;
172  if (poolList.TryGetValue(target, out targetPool))
173  {
174  if (targetPool.Count > 0)
175  {
176  UDEAbstractBullet inst = targetPool.Pop();
177  inst.gameObject.SetActive(true);
178  returnInstance = inst;
179  }
180  else // When there is no more bullet in the pool.
181  {
182  UDEAbstractBullet instance = Instantiate<UDEAbstractBullet>(target, new Vector3(0, 0, 0), Quaternion.Euler(0, 0, 0));
183 #pragma warning disable CS0618 // Type or member is obsolete
184  instance.OriginPrefab = target;
185 #pragma warning restore CS0618 // Type or member is obsolete

187  instObj.name = target.gameObject.name;
188  instObj.transform.parent = this.transform;
189  returnInstance = instance;
190  }
191  }
192  else // When there is no object pool of the target bullet.
193  {
194  Debug.LogWarning("The target bullet is not found. Create new pool for the bullet.");
195  AddNewPool(target, 32);
