174  private void SeedMT()
175  {
176  index = init.n;
177  mt[0] = seed;
178  for (uint i = 1; i < init.n; i++)
179  mt[i] = init.f * (mt[i - 1] ^ (mt[i - 1] >> (w - 2))) + i;
180  }

182  private uint ExtractNumber()
183  {
184  if (index >= init.n)
185  {
186  if (index > init.n)
187  throw new InvalidOperationException("Generator was never seeded");
188 
189  Twist();
190  }
191 
192  uint y = mt[index];
193  y = y ^ ((y >> init.u) & init.d);
194  y = y ^ ((y << init.s) & init.b);
195  y = y ^ ((y << init.t) & init.c);
196  y = y ^ (y >> 1);
197 
198  index++;
199  return y;
200  }

202  private void Twist()
203  {
204  for (uint i = 0; i < init.n; i++)
205  {
206  uint x = (mt[i] & upperMask) + (mt[(i + 1) % init.n] & lowerMask);
207  uint xA = x >> 1;
208  if ((x % 2) != 0)
209  xA = xA ^ init.a;
210  mt[i] = mt[(i + init.m) % init.n] ^ xA;
211  }
212 
213  index = 0;
214  }
