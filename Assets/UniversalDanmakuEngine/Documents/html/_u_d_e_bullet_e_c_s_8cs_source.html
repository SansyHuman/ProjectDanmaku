138  public override void Initialize(Vector2 initPos, Vector2 origin, float initRotation, UDEBaseCharacter originCharacter, UDEBaseShotPattern originShotPattern, UDEBulletMovement[] movements, bool setOriginToCharacter = false, bool loop = false)
139  {
140  if (initialized)
141  {
142  Debug.LogWarning("You tried to initalize bullet that already has initialized. The initialization is ignored.");
143  return;
144  }
145 
146  base.Initialize(initPos, origin, initRotation, originCharacter, originShotPattern, movements, setOriginToCharacter, loop);
147  Translation translation = new Translation() { Value = (Vector3)initPos };
148  RotationEulerXYZ rotation = new RotationEulerXYZ() { Value = new float3(0, 0, math.radians(initRotation)) };
149  if (movements[0].faceToMovingDirection)
150  {
151  switch (movements[0].mode)
152  {
153  case MoveMode.CARTESIAN:
154  rotation.Value.z = UDEMath.Deg(movements[0].velocity);
155  break;
156  case MoveMode.CARTESIAN_POLAR:
157  rotation.Value.z = movements[0].angle;
158  break;
159  case MoveMode.POLAR:
160  rotation.Value.z = movements[0].angularSpeed > 0 ? movements[0].angle + 90 : movements[0].angle - 90;
161  if (movements[0].angularSpeed == 0)
162  rotation.Value.z = movements[0].angle;
163  break;
164  }
165  }
166  UDEBulletMovementECS[] movementsECS = new UDEBulletMovementECS[movements.Length];
167  for (int i = 0; i < movementsECS.Length; i++)
168  movementsECS[i] = movements[i].ToECSMovement();
169  this.movements = movementsECS;
170 
171  UDETime.TimeScale timeScale = UDETime.TimeScale.UNSCALED;
172  if (originCharacter is UDEEnemy)
173  timeScale = UDETime.TimeScale.ENEMY;
174  else if (originCharacter is UDEPlayer)
175  timeScale = UDETime.TimeScale.PLAYER;
176 
177  UDEBulletMovements movement = new UDEBulletMovements()
178  {
179  Movement = this.movements[0],
180  Phase = 0,
181  TotalPhase = this.movements.Length,
182  Time = 0,
183  UsingTimeScale = timeScale,
184  IsSummoning = halo != null,
185  SummonTime = summonTime,
186  Loop = loop
187  };
188 
189  float r = (initPos - origin).magnitude;
190  float angle = r > 0.01f ? UDEMath.Deg(initPos - origin) : movements[0].angle;
191 
192  UDEPolarCoordinate polarCoord = new UDEPolarCoordinate()
193  {
194  Origin = origin,
195  Radius = r,
196  Angle = angle,
197  SetOriginToOriginCharacter = setOriginToCharacter
198  };
199 
200  UDEBulletTimeScale scale = new UDEBulletTimeScale() { Value = UDETime.Instance.GetTimeScale(timeScale) };
201 
202  GameObjectEntity gmObjEntity = originCharacter.gameObject.GetComponent<GameObjectEntity>();
203  if (gmObjEntity == null)
204  gmObjEntity = originCharacter.gameObject.AddComponent<GameObjectEntity>();
205  UDEOriginCharacter originChara = new UDEOriginCharacter() { Value = gmObjEntity.Entity };
206 
207  manager.SetComponentData<Translation>(selfEntity, translation);
208  manager.SetComponentData<RotationEulerXYZ>(selfEntity, rotation);
209  manager.SetComponentData<UDEBulletMovements>(selfEntity, movement);
210  manager.SetComponentData<UDEPolarCoordinate>(selfEntity, polarCoord);
211  manager.SetComponentData<UDEBulletTimeScale>(selfEntity, scale);
212  manager.SetComponentData<UDEOriginCharacter>(selfEntity, originChara);
213 
214  if (movement.IsSummoning)
215  manager.AddComponentData(selfEntity, new UDEBulletSummonPhase());
216 
217  this.originCharacter = originCharacter;
218  this.originShotPattern = originShotPattern;
219  if (originShotPattern != null)
220  _ = originShotPattern + this;
221  transform.SetPositionAndRotation(translation.Value, Quaternion.Euler(0, 0, math.degrees(rotation.Value.z)));
222  }

254  public void ForceMoveToPhase(int phase)
255  {
256  if (phase < 0 || phase > movements.Length - 1)
257  return;
258 
259  UDEBulletMovements movement = manager.GetComponentData<UDEBulletMovements>(selfEntity);
260  if (movement.Phase == phase)
261  return;
262 
263  UDEBulletMovementECS prev = movement.Movement;
264  movement.Movement = movements[phase];
265  UDEPolarCoordinate polarCoord = manager.GetComponentData<UDEPolarCoordinate>(selfEntity);
266  if (movement.Movement.setSpeedToPrevMovement)
267  Management.UDEBulletMovementSystem.PhaseUpdateJob.SyncMovementSpeed(ref prev, ref movement.Movement, ref polarCoord);
268  movement.Phase = phase;
269 
270  manager.SetComponentData<UDEBulletMovements>(selfEntity, movement);
271  }
